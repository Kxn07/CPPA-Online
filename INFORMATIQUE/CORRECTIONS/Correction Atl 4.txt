        int main()
        {
            int n;
            printf("Entrez le nombre de valeurs a saisir : ");
            scanf("%d", &n);
            while (n <= 0)
            {
                printf("Entrez le nombre POSITIF de valeurs a saisir : ");
                scanf("%d", &n);
            }
            int tableau[n];
            printf("Entrez %d valeurs :\n", n);
            for (int i = 0; i < n; i++)
            {
                printf("Valeur %d : ", i + 1);
                scanf("%d", &tableau[i]);
            }

            int max_val = tableau[0];
            int position_max = 0;
            for (int i = 1; i < n; i++)
            {
                if (tableau[i] > max_val)
                {
                    max_val = tableau[i];
                    position_max = i;
                }
            }
            printf("La plus grande valeur est %d et elle se trouve a la position %d .\n", max_val, position_max + 1);

            return 0;
        }

		#include <stdio.h>
        #include <stdlib.h>
        int main()
        {
            int n;
            do
            {
                printf("Entrez la taille du tableau (max 45) : ");
                scanf("%d", &n);

                if (n <= 0 || n > 45)
                {
                    printf("Erreur : la taille doit être entre 1 et 45.\n");
                }
            } while (n <= 0 || n > 45);

            double tableau[n];

            printf("Entrez %d valeurs réelles :\n", n);
            for (int i = 0; i < n; i++)
            {
                printf("Valeur %d : ", i + 1);
                scanf("%lf", &tableau[i]);
            }
            double somme = 0;
            for (int i = 0; i < n; i++)
            {
                somme += tableau[i];
            }
            double moyenne = somme / n;
            int compteur = 0;
            for (int i = 0; i < n; i++)
            {
                if (tableau[i] >= moyenne)
                {
                    compteur++;
                }
            }
            printf("La moyenne des valeurs est : %.2lf\n", moyenne);
            printf("Le nombre de valeurs superieures ou egales a la moyenne est : %d\n", compteur);

            return 0;
        }

	Exercice 6 :
	    #include <stdio.h>
        #include <stdlib.h>

        int main()
        {
            // Demande la taille du tableau
            int t;
            printf("taille=");
            scanf("%d",&t);
            // Vérifie si la taille saisie est valide
            while (t<0 || t>30)
            {
                printf("Warning infErieur a 30 \n");
                scanf("%d",&t);
            }
            // Remplir le tableau
            int tab1[t],i;
            for (i=0;i<t;i++)
            {
                printf("Element d'indice %d =",i);
                scanf("%d",&tab1[i]);
            }
            // Avant classement
            printf("Avant classement");
            for (i=0;i<t;i++)
            {
                printf(" %d",tab1[i]);
            }
            // Apres classement
            int j, max, temp;

            // Tri par sélection en ordre décroissant
            for (i = 0; i < t - 1; i++)
            {
                max = i;
                // Trouver l'élément maximum dans le tableau non trié
                for (j = i + 1; j < t; j++)
                {
                    if (tab1[j] > tab1[max])
                    {
                        max = j;
                    }
                }
                // Échanger l'élément trouvé avec le premier élément de la sous-partie non triée
                temp = tab1[i];
                tab1[i] = tab1[max];
                tab1[max] = temp;
            }

            // Afficher le tableau trié en ordre décroissant
            printf("Tableau trie en ordre decroissant : ");
            for (i = 0; i < t; i++)
            {
                printf("%d ", tab1[i]);
            }

            return 0;
        }
