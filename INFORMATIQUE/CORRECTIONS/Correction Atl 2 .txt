#include <stdio.h>

int main() 
{
    int total_seconds, hours, minutes, seconds;

    // Demande à l'utilisateur de saisir le temps en secondes
    printf("Entrez le temps en secondes: ");
    scanf("%d", &total_seconds);

    // Conversion en heures, minutes et secondes
    hours = total_seconds / 3600;
    minutes = (total_seconds % 3600) / 60;
    seconds = total_seconds % 60;

    // Affichage du résultat
    printf("Temps converti: %d heures, %d minutes, %d secondes\n", hours, minutes, seconds);

    return 0;
}
